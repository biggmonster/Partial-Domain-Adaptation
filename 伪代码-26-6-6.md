Algorithm: CDPDA Training Framework

Input:
    Labeled source domain D_s = {(x_s, y_s)}
    Unlabeled target domain D_t = {x_t}
    Target test set D_test
    Class number C
    Batch size B
    Maximum epoch E
    Mini-batch steps per epoch N = max(number of source batches, number of target batches)
    Loss weights λ_ent, λ_rec, λ_align
    BAA sampling coefficient μ

Output:
    Best adapted model Θ*
    Best target-domain accuracy Acc_best and mean entropy Ent_best

1:  Initialize RepVGG-based feature extractor F
2:  Initialize classifier G
3:  Initialize domain discriminator D
4:  Initialize class2vec encoder E_c2v and decoder D_c2v
5:  Initialize semantic response regressors {M_j}_{j=1}^{d_e}
6:  Initialize optimizer for F, G, D, E_c2v, D_c2v, and {M_j}
7:  Set class-level weight w_c = None
8:  Set Acc_best = 0 and Ent_best = +∞
9:  Set middle domain loader D_m = None

10: for epoch e = 1 to E do
11:     for mini-batch step k = 1 to N do
13:         if e reaches evaluation interval then Evaluate current model on D_test:
15:             For each target test sample x:
16:                 f = F(x)
17:                 p = softmax(G(f))

19:             Compute target-domain accuracy Acc_i
20:             Compute mean prediction entropy Ent_i
21:             Estimate class-level weight:
                    For each class c = 1, ..., C:
22:                     w_c = 1/N_te * (Σ_i p_i_c)

23:             if Acc_i > Acc_best then
24:                 Acc_best = Acc_i
25:                 Ent_best = Ent_i
26:                 Save current model as Θ*
27:             end if

28:             Compute BAA supplement size according to the current epoch:
29:                 rou = max(0,  μ · (1 - 1.5 · e / E))
35:         end if

36:         Sample source batch size B     (x_s, y_s) from D_s
37:         Sample target batch size B     (x_t)      from D_t
39:         Sample middle batch size B*rou (x_m)      from D_s 

41:         Forward into feature extractor F:
42:             f_s = F(x_s), f_t = F(x_t), f_m = F(x_m)                                                   
                Concatenate f_t = [f_t; f_m]
46:             Concatenate f = [f_s; f_t]          
47:                                     
            Forward into ISRA module:
                e =  E_c2v(f)                       
                r =  D_c2v(e)                                                                     
56:             Compute class2vec reconstruction loss:
57:                 L_rec = MSE(f, r)

58:             Train semantic response regressors:
59:                 for each semantic dimension j = 1 to d_e do
60:                     Use g_j regressor to regress the j-th semantic embedding feature e  
61:                     Compute j-th regression loss L_reg_j:
62:                          L_reg_j = MSE(e_j, g_j(f)) + L1
63:                     compute normalized absolute gradient as semantic attention score a_j
64:                         gra_j = |∂L_reg_j / ∂f|
                            a_j = gra_j / (||gra_j||_2 / N)
                        Construct attention-feature masked metrix:
65:                         M_j = a_j ⊙ f
66:                     Split M_j into source and target parts
67:                     Compute j-th semantic response alignment loss:
68:                         L_align_j = MSE(mean(M_s_j), mean(M_t_j))
                Compute semantic response alignment loss L_align:
70:                 L_align = Σ_j L_align_j

            Forward into classifier G: 
                Compute source weighted classification cross-entropy loss L_cls:        
73:                 L_cls = weighted_CE(G(f_s), y_  s, w_c)       
77:             Compute target mean entropy loss L_ent:
78:                 L_ent = mean Entropy(softmax(G(f_t)))

            Forward into domain discriminator D: 
79:             Compute entropy-aware adversarial domain loss:
81:                 H = Entropy(softmax(G(f)))
82:                 L_adv = weighted_DANN_loss(f, D, H, w_c)

83:         Compute total objective:
84:             L_total = L_cls
85:                     + λ_ent · L_ent
86:                     + L_adv
87:                     + λ_rec · L_rec
88:                     + λ_align · L_align

89:         Backpropagate L_total
90:         Update F, G, D, E_c2v, R_c2v, and {M_j}

91:     end for
92: end for

93: return Θ*, Acc_best, Ent_best
